001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Transitions an input position of a Levenshtein state to all candidate
005 * positions, conditioned by a set of parameters.
006 * @author Dylon Edwards
007 * @since 2.1.0
008 */
009public interface IPositionTransitionFunction {
010
011  /**
012   * Returns a state with all the possible transitions from the current
013   * position.
014   * @param n Maximum number of errors to tolerate in spelling candidates.
015   * @param position Vector consisting of the current index of the spelling
016   * candidate, the number of errors accumulated up to that index, and
017   * (optionally) a flag specifying whether the position is a special kind as
018   * defined by its specific, Levenshtein algorithm (e.g. whether it is a
019   * transposition position, etc.).
020   * @param characteristicVector Relevant subwords consisting of booleans about
021   * whether the characters associated by their indices in the spelling
022   * candidate are the same as the character being sought from the query term.
023   * @param offset Offset for various operations within the transition function.
024   * @return New state consisting of all possible transitions for the given
025   * position, given the other paramters.
026   */
027  IState of(int n, int[] position, boolean[] characteristicVector, int offset);
028}
