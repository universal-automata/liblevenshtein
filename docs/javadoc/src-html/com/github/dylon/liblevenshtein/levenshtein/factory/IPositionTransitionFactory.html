001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import com.github.dylon.liblevenshtein.levenshtein.IPositionTransitionFunction;
004
005/**
006 * Returns functions that transition positions to all possible positions, given
007 * the evidence.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public interface IPositionTransitionFactory {
012
013  /**
014   * Builds a new position-transition function.
015   * @return New position-transition function.
016   */
017  IPositionTransitionFunction build();
018
019  /**
020   * Recycles a position-transition function for re-use.
021   * @param transition {@link IPositionTransitionFunction} to recycle.  You must
022   * discard your reference to {@code transition} once it's been recycled.
023   */
024  default void recycle(IPositionTransitionFunction transition) {
025    // default behavior is to do nothing
026  }
027}
