001package com.github.dylon.liblevenshtein.levenshtein.distance;
002
003import lombok.NonNull;
004
005import it.unimi.dsi.fastutil.objects.Object2IntMap;
006import it.unimi.dsi.fastutil.objects.Object2IntOpenHashMap;
007
008import com.github.dylon.liblevenshtein.collection.SymmetricImmutablePair;
009import com.github.dylon.liblevenshtein.levenshtein.IDistance;
010
011/**
012 * Common, initialization logic for memoized, distance metrics.
013 * @author Dylon Edwards
014 * @since 2.1.0
015 */
016public abstract class AbstractMemoized implements IDistance<String> {
017
018  /** Default return value of memoized distances when no record exists. */
019  protected static final int DEFAULT_RETURN_VALUE = -1;
020
021  /** Memoizes the distance pairs of terms. */
022  protected final Object2IntMap<SymmetricImmutablePair<String>> memo;
023
024  /**
025   * Initializes the memoization map, etc.
026   */
027  public AbstractMemoized() {
028    memo = new Object2IntOpenHashMap<SymmetricImmutablePair<String>>();
029    memo.defaultReturnValue(DEFAULT_RETURN_VALUE);
030  }
031
032  /**
033   * If {@code t} is less than the length of {@code u}, then return the
034   * substring of {@code u} beginning at {@code 1 + t}; Otherwise, return an
035   * empty string. It is assumed that {@code u} and {@code v} are not null.
036   * @param u String to slice
037   * @param t Lower bound of the index at which to slice {@code u}
038   * @return The substring of {@code u} beginning at index, {@code 1 + t}, if
039   * {@code t} is less than the length of {@code u}, or an empty string
040   * otherwise.
041   */
042  public String f(final String u, final int t) {
043    if (t < u.length()) {
044      return u.substring(1+t);
045    }
046
047    return "";
048  }
049
050  /**
051   * {@inheritDoc}
052   */
053  @Override
054  public int between(@NonNull final String v, @NonNull final String w) {
055    // Don't want to check for nullity on each recursion ...
056    return memoizedDistance(v, w);
057  }
058
059  /**
060   * Memoizes the distance between terms, {@code v} and {@code w}, and returns
061   * it.
062   * @param v Term to compare with {@code w}
063   * @param w Term to compare with {@code v}
064   * @return Distance between {@code v} and {@code w}
065   */
066  public abstract int memoizedDistance(String v, String w);
067}
