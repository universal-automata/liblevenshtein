001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.util.Comparator;
004
005/**
006 * Defines the methods on Levenshtein states, which consist of all possible
007 * position vectors at the current index of the spelling candidate.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public interface IState {
012
013  /**
014   * Number of positions in this state.
015   * @return Number of positions in this state.
016   */
017  int size();
018
019  /**
020   * Adds a new position to the linked-list of positions in this state.
021   * @param position Position to add to those already in this state.
022   */
023  void add(int[] position);
024
025  /**
026   * Inserts a position into a specific location of the linked-list of positions.
027   * @param index Location at which to insert the position.
028   * @param position Position vector to insert into this state.
029   */
030  void insert(int index, int[] position);
031
032  /**
033   * One of two cursors maintained in this state, which returns the position at
034   * the requested index.  Subsequent requests for positions at the immediately
035   * proceeding or preceding indices will begin at the index of the last
036   * position requested, so lookup for consecutive positions is fast.
037   * <p>
038   * {@link #getOuter(int)} is used within the outer loops of
039   * {@link com.github.dylon.liblevenshtein.levenshtein.UnsubsumeFunction}s.
040   * @param index Location of the position to return
041   * @return Position vector at the requested index.
042   * @see #getInner(int)
043   */
044  int[] getOuter(int index);
045
046  /**
047   * One of two cursors maintained in this state, which returns the position at
048   * the requested index.  Subsequent requests for positions at the immediately
049   * proceeding or preceding indices will begin at the index of the last
050   * position requested, so lookup for consecutive positions is fast.
051   * <p>
052   * {@link #getInner(int)} is used within the inner loops of
053   * {@link com.github.dylon.liblevenshtein.levenshtein.UnsubsumeFunction}s.
054   * @param index Location of the position to return
055   * @return Position vector at the requested index.
056   * @see #getOuter(int)
057   */
058  int[] getInner(int index);
059
060  /**
061   * Unlinks the position corresponding to the inner cursor, at the last index
062   * requested of {@link #getInner(int)}.
063   * @return Position vector that was unlinked from those in this state.
064   */
065  int[] removeInner();
066
067  /**
068   * Unlinks and recycles all the positions in this state, making it empty. This
069   * is usually done just before this state is itself recycled.
070   */
071  void clear();
072
073  /**
074   * Merge-sorts the positions in this state in a fashion that makes
075   * un-subsumption easy.
076   * @param comparator Describes how to sort the positions (dependent on the
077   * Levenshtein algorithm).
078   */
079  void sort(Comparator<int[]> comparator);
080}
