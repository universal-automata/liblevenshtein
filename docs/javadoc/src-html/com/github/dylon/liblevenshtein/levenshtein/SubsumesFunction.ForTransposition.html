001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Routines for determining whether one position subsumes another.
005 * @author Dylon Edwards
006 * @since 2.1.0
007 */
008public abstract class SubsumesFunction implements ISubsumesFunction {
009
010  /**
011   * {@inheritDoc}
012   */
013  @Override
014  public boolean at(
015      final int i, final int e,
016      final int j, final int f) {
017
018    throw new UnsupportedOperationException("at(i,e, j,f) is not supported");
019  }
020
021  /**
022   * {@inheritDoc}
023   */
024  @Override
025  public boolean at(
026      final int i, final int e, final int s,
027      final int j, final int f, final int t,
028      final int n) {
029
030    throw new UnsupportedOperationException(
031        "at(i,e,s, j,f,t, n) is not supported");
032  }
033
034  /**
035   * Routines for determining whether a standard position subsumes another.
036   * @author Dylon Edwards
037   * @since 2.1.0
038   */
039  public static class ForStandardAlgorithm extends SubsumesFunction {
040
041    /**
042     * {@inheritDoc}
043     */
044    @Override
045    public boolean at(
046        final int i, final int e,
047        final int j, final int f) {
048
049      return ((i < j) ? (j - i) : (i -j)) <= (f - e);
050    }
051  }
052
053  /**
054   * Routines for determining whether a transposition position subsumes another.
055   * @author Dylon Edwards
056   * @since 2.1.0
057   */
058  public static class ForTransposition extends SubsumesFunction {
059
060    /**
061     * {@inheritDoc}
062     */
063    @Override
064    public boolean at(
065        final int i, final int e, final int s,
066        final int j, final int f, final int t,
067        final int n) {
068
069      if (s == 1) {
070        if (t == 1) {
071          return (i == j);
072        }
073
074        return (f == n) && (i == j);
075      }
076
077      if (t == 1) {
078        // We have two cases:
079        //
080        // Case 1: (j < i) => (j - i) = - (i - j)
081        //                 => |j - (i - 1)| = |j - i + 1|
082        //                                  = |-(i - j) + 1|
083        //                                  = |-(i - j - 1)|
084        //                                  = i - j - 1
085        //
086        // Case 1 holds, because i and j are integers, and j < i implies i is at
087        // least 1 unit greater than j, further implying that i - j - 1 is
088        // non-negative.
089        //
090        // Case 2: (j >= i) => |j - (i - 1)| = |j - i + 1| = j - i + 1
091        //
092        // Case 2 holds for the same reason case 1 does, in that j - i >= 0, and
093        // adding 1 to the difference will only strengthen its non-negativity.
094        //
095        //return Math.abs(j - (i - 1)) <= (f - e);
096        return ((j < i) ? (i - j - 1) : (j - i + 1)) <= (f - e);
097      }
098
099      return ((i < j) ? (j - i) : (i - j)) <= (f - e);
100    }
101  }
102
103  /**
104   * Routines for determining whether a merge-and-split position subsumes another.
105   * @author Dylon Edwards
106   * @since 2.1.0
107   */
108  public static class ForMergeAndSplit extends SubsumesFunction {
109
110    /**
111     * {@inheritDoc}
112     */
113    @Override
114    public boolean at(
115        final int i, final int e, final int s,
116        final int j, final int f, final int t,
117        final int n) {
118
119      if (s == 1 && t == 0) {
120        return false;
121      }
122
123      return ((i < j) ? (j - i) : (i - j)) <= (f - e);
124    }
125  }
126}
