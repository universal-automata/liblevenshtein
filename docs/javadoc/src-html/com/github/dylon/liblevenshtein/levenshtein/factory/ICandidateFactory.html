001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003/**
004 * Builds spelling candidates of the requested type, optionally including the
005 * distance of the candidate from the query term.
006 * @param <CandidateType> Kind of spelling candidate built by this factory.
007 * @author Dylon Edwards
008 * @since 2.1.2
009 */
010public interface ICandidateFactory<CandidateType> {
011
012  /**
013   * Builds a new spelling candidate from the dictionary term and its
014   * Levenshtein distance from the query term.
015   * @param term Candidate term from the dictionary.
016   * @param distance Levenshtein distance of the dictionary term from the query
017   * term.
018   * @return A new spelling candidate, optionally with the distance included.
019   */
020  CandidateType build(String term, int distance);
021}
