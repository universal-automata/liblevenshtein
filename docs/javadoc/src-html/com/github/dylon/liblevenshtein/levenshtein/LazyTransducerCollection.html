139    // f(x) := x * 2 + 1
140    // a := (n - 1) / 2
141    // f(a) = (n - 1) / 2 * 2 + 1 = n - 1 + 1 = n
142    //
143    // We want to cap the value of "a" at "n = max integer" so "f(a)" does not
144    // overflow, which it would if "a > (n - 1) / 2".  In other words, define:
145    // g(x) := { x * 2 + 1  , if a < (n - 1) / 2
146    //         { n          , otherwise
147    //
148    // If there was no upper bound for integer values, this would be equivalent:
149    // h(x) := min {f(x), n}
150    this.a = (maxDistance < (Integer.MAX_VALUE - 1) >> 1) ? (maxDistance << 1) + 1 : Integer.MAX_VALUE;
