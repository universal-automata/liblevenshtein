001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import com.github.dylon.liblevenshtein.levenshtein.Candidate;
004
005/**
006 * Builds spelling candidates of the requested type, optionally including the
007 * distance of the candidate from the query term.
008 * @param <CandidateType> Kind of spelling candidate built by this factory.
009 * @author Dylon Edwards
010 * @since 2.1.2
011 */
012public abstract class CandidateFactory<CandidateType>
013    implements ICandidateFactory<CandidateType> {
014
015  /**
016   * Builds instances of {@link Candidate}, with the dictionary term and its
017   * Levenshtein distance from the query term.
018   * @author Dylon Edwards
019   * @since 2.1.2
020   */
021  public static class WithDistance extends CandidateFactory<Candidate> {
022
023    /**
024     * {@inheritDoc}
025     */
026    public Candidate build(final String term, final int distance) {
027      return new Candidate(term, distance);
028    }
029  }
030
031  /**
032   * Does not include the Levenshtein distance, but returns the dictionary term
033   * alone.
034   * @author Dylon Edwards
035   * @since 2.1.2
036   */
037  public static class WithoutDistance extends CandidateFactory<String> {
038
039    /**
040     * {@inheritDoc}
041     */
042    public String build(final String term, final int distance) {
043      return term;
044    }
045  }
046}
