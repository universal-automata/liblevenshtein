001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Determines whether the position represented by the combination of parameters
005 * in the left half subsumes the position represented by the combination of
006 * parameters in the right half.
007 * @author Dylon Edwards
008 * @since 2.1.0
009 */
010public interface ISubsumesFunction {
011
012  /**
013   * Determines whether the standard, Levenshtein state represented by
014   * {@code (i,e)} subsumes the other, represented by {@code (j,f)}.
015   * @param i Index of the spelling candidate of the first position.
016   * @param e Number of accumulated errors of the spelling candidate at index
017   * {@code i} of the first position.
018   * @param j Index of the spelling candidate of the second position.
019   * @param f Number of accumulated errors of the spelling candidate at index
020   * {@code j} of the second position.
021   * @return Whether {@code (i,e)} subsumes {@code (j,f)}
022   */
023  boolean at(int i, int e, int j, int f);
024
025  /**
026   * Determines whether the transposition or merge-and-split, Levenshtein state
027   * represented by {@code (i,e,s)} subsumes the other, represented by
028   * {@code (j,f,t)}, under the maximum number of errors, {@code n}.
029   * @param i Index of the spelling candidate of the first position.
030   * @param e Number of accumulated errors of the spelling candidate at index
031   * {@code i} of the first position.
032   * @param s Either {@code 0} or {@code 1}, depending on whether the first
033   * position is a regular state or a special case (e.g. transposition, merge,
034   * or split).
035   * @param j Index of the spelling candidate of the second position.
036   * @param f Number of accumulated errors of the spelling candidate at index
037   * {@code j} of the second position.
038   * @param t Either {@code 0} or {@code 1}, depending on whether the second
039   * position is a regular state or a special case (e.g. transposition, merge,
040   * or split).
041   * @param n Maximum number of errors allowed in spelling candidates.
042   * @return Whether {@code (i,e,s)} subsumes {@code (j,f,t)}
043   */
044  boolean at(int i, int e, int s, int j, int f, int t, int n);
045}
