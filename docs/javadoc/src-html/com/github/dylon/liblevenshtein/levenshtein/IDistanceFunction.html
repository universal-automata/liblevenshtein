001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * The distance of each position in a state can be defined as follows:
005 * <p>
006 * <code>distance = w - i + e</code>
007 * <p>
008 * For every accepting position, it must be the case that <code>w - i &le; n - e</code>.
009 * It follows directly that the distance of every accepted position must be no
010 * more than {@code n}:
011 * <p>
012 * <code>(w - i &le; n - e) &hArr; (w - i + e &le; n) &hArr; (distance &le; n)</code>
013 * <p>
014 * The Levenshtein distance between any two terms is defined as the minimum
015 * edit distance between the two terms.  Therefore, iterate over each position
016 * in an accepting state, and take the minimum distance among all its accepting
017 * positions as the corresponding Levenshtein distance.
018 * @author Dylon Edwards
019 * @since 2.1.0
020 */
021public interface IDistanceFunction {
022
023  /**
024   * @param state Levenshtein state whose minimum, Levenshtein distance is to be
025   * determined.
026   * @param w Length of the query term.
027   * @return Minimum distance between the spelling candidate and query term,
028   * implied by the positions within {@code state}.
029   */
030  int at(IState state, int w);
031}
