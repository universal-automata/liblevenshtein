001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004import java.util.Comparator;
005
006/**
007 * Sorts position vectors for the standard, Levenshtein algorithm.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public class StandardPositionComparator implements Serializable, Comparator<int[]> {
012  private static final long serialVersionUID = 1L;
013
014  /**
015   * {@inheritDoc}
016   */
017  @Override
018  public int compare(final int[] a, final int[] b) {
019    int c = a[0] - b[0];
020    if (0 != c) return c;
021    return a[1] - b[1];
022  }
023}
