001package com.github.dylon.liblevenshtein.collection.dawg.factory;
002
003import com.github.dylon.liblevenshtein.collection.dawg.IDawgNode;
004import com.github.dylon.liblevenshtein.collection.dawg.Transition;
005
006/**
007 * Builds transitions that link nodes together, connected by a character label.
008 * @param <NodeType> Kind of the nodes joined by transitions.
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface ITransitionFactory<NodeType extends IDawgNode<NodeType>> {
013
014  /**
015   * Builds or recycles a {@link Transition} object representing an edge from
016   * {@code source} to {@code target}, annotated with {@code label}.
017   * @param source Node from which the transition is leaving.
018   * @param label Label that joins {@code source} and {@code target} together.
019   * @param target Node to which the transition is going.
020   * @return A {@link Transition} linking {@code source} to {@code target},
021   * under {@code label}.
022   */
023  Transition<NodeType> build(NodeType source, char label, NodeType target);
024
025  /**
026   * Recycles a {@link Transition} to be re-used from
027   * {@link #build(IDawgNode,char,IDawgNode) build}.
028   * @param transition {@link Transition} to recycle.  Note that once the
029   * transition has been recycled, you should discard it (like a deleted
030   * reference in C++).
031   */
032  default void recycle(Transition<NodeType> transition) {
033    // default behavior is to do nothing
034  }
035}
