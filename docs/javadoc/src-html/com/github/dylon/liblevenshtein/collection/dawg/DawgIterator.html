001package com.github.dylon.liblevenshtein.collection.dawg;
002
003import java.util.Queue;
004import java.util.ArrayDeque;
005
006import lombok.NonNull;
007
008import it.unimi.dsi.fastutil.chars.CharIterator;
009
010import com.github.dylon.liblevenshtein.collection.dawg.factory.IPrefixFactory;
011import com.github.dylon.liblevenshtein.collection.AbstractIterator;
012
013/**
014 * Iterates over the terms within an {@link AbstractDawg}.
015 * @author Dylon Edwards
016 * @since 2.1.0
017 */
018public class DawgIterator extends AbstractIterator<String> {
019
020  /**
021   * Queue for traversing the terms in the {@link AbstractDawg} in a
022   * depth-first search manner.
023   */
024  private final Queue<Prefix<DawgNode>> prefixes = new ArrayDeque<>();
025
026  /**
027   * Returns whether the current {@link DawgNode} represents the last character
028   * in some term.
029   */
030  private final IFinalFunction<DawgNode> isFinal;
031
032  /** Creates and caches {@link Prefix} instances. */
033  private final IPrefixFactory<DawgNode> prefixFactory;
034
035  /**
036   * Initializes a new {@link DawgIterator} with a {@link IPrefixFactory} and an
037   * {@link AbstractDawg}.
038   * @param prefixFactory Creates and caches {@link Prefix} instances, which are
039   * used to traverse the {@code dawg}.
040   * @param dawg {@link AbstractDawg} to iterate over.
041   */
042  public DawgIterator(
043      @NonNull final IPrefixFactory<DawgNode> prefixFactory,
044      @NonNull final AbstractDawg dawg) {
045    this.isFinal = dawg;
046    this.prefixFactory = prefixFactory;
047    prefixes.offer(prefixFactory.build(dawg.root(), ""));
048  }
049
050  /**
051   * {@inheritDoc}
052   */
053  @Override
054  protected void advance() {
055    if (null == next && !prefixes.isEmpty()) {
056      DawgNode node = null;
057      String value = null;
058
059      do {
060        final Prefix<DawgNode> prefix = prefixes.poll();
061        node = prefix.node();
062        value = prefix.value();
063        final CharIterator iter = node.labels();
064        while (iter.hasNext()) {
065          final char label = iter.nextChar();
066          final DawgNode nextNode = node.transition(label);
067          final String nextValue = value + label;
068          if (!prefixes.offer(prefixFactory.build(nextNode, nextValue))) {
069            throw new IllegalStateException(
070                "Failed to enqueue prefix value: " + nextValue);
071          }
072        }
073      }
074      while (!isFinal.at(node));
075      this.next = value;
076    }
077  }
078}
