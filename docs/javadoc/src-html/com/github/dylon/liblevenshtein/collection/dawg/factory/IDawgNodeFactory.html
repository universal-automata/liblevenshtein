001package com.github.dylon.liblevenshtein.collection.dawg.factory;
002
003import it.unimi.dsi.fastutil.chars.Char2ObjectMap;
004
005import com.github.dylon.liblevenshtein.collection.dawg.IDawgNode;
006
007/**
008 * Builds nodes for use in DAWG structures. Implementations may do such things
009 * as manage object pools, etc.
010 * @param <NodeType> Kind of node built in this factory.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public interface IDawgNodeFactory<NodeType extends IDawgNode<NodeType>> {
015
016  /**
017   * Builds a DAWG node
018   * @return A DAWG node
019   * @throws NullPointerException When edges is null
020   */
021  NodeType build();
022
023  /**
024   * [Green Operation] In an effort to fight global warming and environmental
025   * polution, you may invoke this method when you no longer have need of a node
026   * so it may be recycled for future use.
027   * @param node Node to recycle
028   * @return This factory, for chaining method calls.
029   */
030  IDawgNodeFactory<NodeType> recycle(NodeType node);
031}
